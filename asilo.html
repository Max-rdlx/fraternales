} }); } // Datos de las imágenes para el lightbox const galleryItems = Array.from(document.querySelectorAll('.gallery-item')); let visibleItems = galleryItems.slice(); // Filtros const filterBtns = document.querySelectorAll('.filter-btn'); filterBtns.forEach(btn => { btn.addEventListener('click', () => { filterBtns.forEach(b => b.classList.remove('active')); btn.classList.add('active'); const filter = btn.dataset.filter; visibleItems = []; galleryItems.forEach((item) => { const category = item.dataset.category; const matches = (filter === 'all' || category === filter); if (matches) { item.classList.remove('hidden'); visibleItems.push(item); item.style.animation = 'none'; item.offsetHeight; item.style.animation = `fadeInUp 0.6s cubic-bezier(0.4, 0, 0.2, 1) forwards`; item.style.animationDelay = `${visibleItems.length * 0.05}s`; } else { item.classList.add('hidden'); item.style.animation = ''; } }); }); }); // Lightbox const lightbox = document.getElementById('lightbox'); if (!lightbox) { console.error('Lightbox element not found'); return; } const lightboxImg = lightbox.querySelector('.lightbox-img'); const lightboxTitle = lightbox.querySelector('.lightbox-title'); const lightboxCategory = lightbox.querySelector('.lightbox-category'); const lightboxCounter = lightbox.querySelector('.lightbox-counter'); let currentIndex = 0; function updateVisibleItems() { visibleItems = galleryItems.filter(item => !item.classList.contains('hidden')); } function openLightbox(index) { updateVisibleItems(); if (visibleItems.length === 0) return; if (index < 0 || index >= visibleItems.length) return; const item = visibleItems[index]; const img = item.querySelector('img'); const title = item.querySelector('h3').textContent; const category = item.querySelector('p').textContent; currentIndex = index; lightbox.classList.add('loading'); lightbox.classList.add('active'); lightboxImg.src = ''; const newImg = new Image(); newImg.onload = () => { lightboxImg.src = img.src; lightboxTitle.textContent = title; lightboxCategory.textContent = category; lightboxCounter.textContent = `${index + 1} / ${visibleItems.length}`; lightbox.classList.remove('loading'); }; newImg.onerror = () => { lightboxTitle.textContent = 'Error al cargar imagen'; lightbox.classList.remove('loading'); }; newImg.src = img.src; document.body.style.overflow = 'hidden'; } function closeLightbox() { lightbox.classList.remove('active'); document.body.style.overflow = ''; setTimeout(() => { lightboxImg.src = ''; }, 300); } function navigateLightbox(direction) { if (visibleItems.length === 0) return; let newIndex = currentIndex + direction; if (newIndex < 0) newIndex = visibleItems.length - 1; if (newIndex >= visibleItems.length) newIndex = 0; openLightbox(newIndex); } // Event listeners para abrir lightbox - CORREGIDO galleryItems.forEach((item) => { // Click en toda la tarjeta item.addEventListener('click', (e) => { // No abrir si se hizo click en un enlace if (e.target.closest('a')) return; updateVisibleItems(); const visibleIndex = visibleItems.indexOf(item); if (visibleIndex !== -1) { openLightbox(visibleIndex); } }); // Click específico en el botón "Ver Detalle" const viewBtn = item.querySelector('.view-btn'); if (viewBtn) { viewBtn.addEventListener('click', (e) => { e.stopPropagation(); // Evitar que se dispare el click de la tarjeta updateVisibleItems(); const visibleIndex = visibleItems.indexOf(item); if (visibleIndex !== -1) { openLightbox(visibleIndex); } }); } }); // Controles del lightbox const closeBtn = lightbox.querySelector('.lightbox-close'); const prevBtn = lightbox.querySelector('.prev'); const nextBtn = lightbox.querySelector('.next'); if (closeBtn) closeBtn.addEventListener('click', closeLightbox); if (prevBtn) prevBtn.addEventListener('click', (e) => { e.stopPropagation(); navigateLightbox(-1); }); if (nextBtn) nextBtn.addEventListener('click', (e) => { e.stopPropagation(); navigateLightbox(1); }); lightbox.addEventListener('click', (e) => { if (e.target === lightbox || e.target.closest('.lightbox-content') === null) { closeLightbox(); } }); document.addEventListener('keydown', (e) => { if (!lightbox.classList.contains('active')) return; switch(e.key) { case 'Escape': closeLightbox(); break; case 'ArrowLeft': e.preventDefault(); navigateLightbox(-1); break; case 'ArrowRight': e.preventDefault(); navigateLightbox(1); break; } }); const scrollIndicator = document.querySelector('.scroll-indicator'); const filters = document.querySelector('.filters'); if (scrollIndicator && filters) { scrollIndicator.addEventListener('click', () => { filters.scrollIntoView({ behavior: 'smooth' }); }); } const imageObserver = new IntersectionObserver((entries, observer) => { entries.forEach(entry => { if (entry.isIntersecting) { const img = entry.target; img.classList.add('loaded'); observer.unobserve(img); } }); }, { rootMargin: '50px' }); document.querySelectorAll('.gallery-item img').forEach(img => { imageObserver.observe(img); }); let ticking = false; window.addEventListener('scroll', () => { if (!ticking) { requestAnimationFrame(() => { const scrolled = window.pageYOffset; const header = document.querySelector('.header-content'); if (header) { header.style.transform = `translateY(${scrolled * 0.5}px)`; header.style.opacity = Math.max(0, 1 - (scrolled / 700)); } ticking = false; }); ticking = true; } }, { passive: true }); document.body.addEventListener('dragstart', e => { if (e.target.tagName === 'IMG') e.preventDefault(); }); });
